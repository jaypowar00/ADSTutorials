Q.  Write A PL/SQL Procedure To Reverse A Given Number (MySQL).


> DELIMITER $$

> DROP PROCEDURE IF EXISTS reverse_number$$

> CREATE PROCEDURE reverse_number(IN num INT)
  BEGIN
  DECLARE r_num INT;
  DECLARE n1 INT;
  SET r_num=0;
  SET n1 = num;
  WHILE n1 > 9 DO
  SET r_num = (r_num * 10) + (n1 mod 10);
  SET n1 = n1 / 10;
  END WHILE;
  SET r_num = (r_num * 10) + (n1 mod 10);
  SELECT num AS Number, r_num AS "Reverse Number";
  END$$

> DELIMITER ;




> CALL reverse_number(123);
